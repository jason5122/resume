% Custom commands
\newcommand{\resumeheading}[3]{
    \vspace{-1pt}
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}
        \textbf{#1} & {\small#2}\\
        \textsl{\small#3}
    \end{tabular*}
}

\newcommand{\personalheading}[3]{
    \vspace{-1pt}
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}
        \textbf{#1} \textsl{\small#2} &
        {\small#3}
    \end{tabular*}
}
